018package org.apache.commons.daemon;
019
020
021/**
022 * Defines a set of methods that a Daemon instance can use to
023 * communicate with the Daemon container.
024 *
025 * @author Pier Fumagalli
026 * @version $Id: DaemonContext.java 1204006 2011-11-19 16:09:15Z ggregory $
027 */
028public interface DaemonContext
029{
030
031    /**
032     * Returns  <code>DaemonController</code> object that can be used
033     * to control the <code>Daemon</code> instance that this
034     * <code>DaemonContext</code> is passed to.
035     */
036    public DaemonController getController();
037
038    /**
039     * Returns an array of <code>String</code> arguments supplied by
040     * the environment.  corresponding to the array of arguments given
041     * in the <code>public static void main()</code> method used as an
042     * entry point to most other java programs.
043     */
044    public String[] getArguments();
045
046}
